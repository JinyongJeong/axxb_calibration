1function [xi_hat] = twistlog(h)
 2%TWISTLOG  calculate the log of a homogeneous transformation
 3%
 4%   [XI_HAT] = TWISTLOG(H)
 5%
 6% See also: TWIST, SKEWEXP, TWISTEXP, SKEWLOG.
 7
 8% $Id: twistlog.m,v 1.1 2009-03-17 16:40:18 bradleyk Exp $
 9% Copyright (C) 2005, by Brad Kratochvil
10
11% this algorithm from Murray pg. 414
12
13% Commented out for .mex
14% global DebugLevel;
15% 
16% if isempty(DebugLevel) || (DebugLevel > 1)
17%   if ~isrot(h(1:3,1:3)),
18%     error('ROBOTLINKS:skewlog','H is not a homogeneous transformation')
19%   end
20% end
21
22w_hat = skewlog(h(1:3, 1:3));
23w = skewcoords(w_hat);
24
25w_norm = norm(w);
26p = h(1:3, 4);
27
28if isequalf(zeros(3,1), p),
29  A_inv = zeros(3,3);
30elseif 0 == sin(w_norm) || 0 == w_norm,
31  A_inv = eye(3);
32else
33  A_inv = eye(3) - ...
34          w_hat/2 + ...
35          (2*sin(w_norm)-w_norm*(1+cos(w_norm)))/(2*w_norm^2*sin(w_norm))*w_hat*w_hat;
36end
37
38xi_hat = [w_hat A_inv*p; 0 0 0 0];
39theta = w_norm;
40
41end
