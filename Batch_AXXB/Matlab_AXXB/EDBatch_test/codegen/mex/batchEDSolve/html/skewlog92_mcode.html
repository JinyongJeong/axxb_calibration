1function [w_hat] = skewlog(R)
 2%SKEWLOG  calculate the log of a rotation matrix
 3%
 4%   [W_HAT] = SKEWLOG(R)
 5%
 6%
 7% See also: TWIST, SKEWEXP, TWISTEXP, TWISTLOG.
 8
 9% $Id: skewlog.m,v 1.1 2009-03-17 16:40:18 bradleyk Exp $
10% Copyright (C) 2005, by Brad Kratochvil
11
12% this algorithm from Murray pg. 414
13
14% Commented out for .mex
15% global DebugLevel;
16% 
17% if isempty(DebugLevel) || (DebugLevel > 1)
18%   if ~isrot(R),
19%     error('ROBOTLINKS:skewlog','R is not a rotation matrix')
20%   end
21% end
22
23if isequalf(R, eye(3), 1e-6),
24  theta = 0;
25  w_hat = zeros(3,3);
26else
27  val = (trace(R)-1)/2;
28  % this is to clamp the values to +-1, which sometimes
29  % doesn't happen due to floating point noise
30  if val > 1,
31    val = 1;
32  elseif val < -1,
33    val = -1;
34  end  
35  theta = acos(val);
36  if 0 == theta,
37    w_hat = zeros(3,3);
38  else
39    w_hat = (R-R')/(2*sin(theta))*theta;
40  end
41end
