1%Function to extract Plucker parameters from a g \in SE(3)
 2
 3%files needed:
 4%so3_vec().m
 5
 6function [theta, N, d, p] = param_extract(X_in)
 7
 8X = zeros(4,4);
 9X = X_in;
10
11%extract theta---------------------------------
12theta=acos((trace(X(1:3,1:3))-1)/2);
13%----------------------------------------------
14
15%extract N---------------------------------
16N=(X(1:3,1:3)-X(1:3,1:3)')/(2*sin(theta));
17%----------------------------------------------
18
19%extract d---------------------------------
20d=dot(X(1:3,4),so3_vec(N));
21%----------------------------------------------
22
23%extract p---------------------------------
24n=so3_vec(N);
25u=1/sqrt(n(1)^2+n(2)^2)*[-n(2); n(1); 0];
26c=linsolve([1-cos(theta) sin(theta); -sin(theta) 1-cos(theta)], [dot(X(1:3,4),u); dot(X(1:3,4),(cross(n,u)))]);
27
28p=c(1)*u+c(2)*cross(n,u);
29%----------------------------------------------
30
31end
