1% This function calculates the 1st order approximation of the mean of a
 2% bunch of matrices based on the Taylor expansion of the matrix logarithm
 3% and the definition of mean of a probability density function.
 4
 5% Input: X is a 4 by 4*n rigid transformation matrices
 6% Output: M_T1 is the mean of the 1st order approximation of Taylor
 7% expansion
 8
 9% Change of this m file doesn't automatically change the executable generated by 
10% mean_Taylor_2nd.m
11function [M_1, M_hat] = mean_Taylor_1st( X ) %#codegen
12
13n =  size(X, 2)/4;
14
15M_hat = zeros(4);
16M_1 = zeros(4);
17
18for i = 1:n
19    M_hat = M_hat + X(:,(i-1)*4+1:i*4);
20end
21
22M_hat = 1/n*M_hat;  % Note that M_hat doesn't belong to SE(3)
23
24M_1 = orthog(M_hat);
25
26end
