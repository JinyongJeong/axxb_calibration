1function [b] = isequalf(a, b, thresh)
 2%ISEQUALF  Returns true if the two quantities are equal within a threshold
 3%
 4%   t = ISEQUALF(A, B)
 5%   [t reason] = ISEQUALF(A, B, THRESH)
 6%
 7% This function is useful for floating point values, where there may be
 8% a small difference.
 9%
10% See also: ISEQUAL.
11
12% $Id: isequalf.m,v 1.1 2009-03-17 16:40:18 bradleyk Exp $
13% Copyright (C) 2005, by Brad Kratochvil
14
15if isa(a, 'sym') || isa(b, 'sym'),
16  b = isequal(a,b);
17else
18
19  if 2 == nargin,
20    thresh = eps()*100; % it's machine dependent now
21  end
22
23  m = max(abs(a-b));
24
25  if any(m > thresh)
26    b = false;
27  else
28    b = true;
29  end
30
31end
